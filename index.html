-- Another analytical query: Customers with multiple reservations
SELECT   
  C.CUS_FNAME || ' ' || C.CUS_LNAME AS FULL_NAME,  
  C.CUS_EMAIL,  
  COUNT(R.RES_ID) AS RESERVATION_COUNT  
FROM UNTP019.CUSTOMER C  
JOIN UNTP019.RESERVE R ON C.CUS_ID = R.CUS_ID  
GROUP BY C.CUS_ID, C.CUS_FNAME, C.CUS_LNAME, C.CUS_EMAIL

Mean Green EagleLift Ride-Sharing System ‚Äì C++ (Fall 2023)

Designed a C++ ride-sharing system with ERD for data modeling and matching algorithms for efficiency. Analyzed simulated ride data for inclusivity metrics (e.g., accessibility matches)‚Äîdemonstrating data-driven optimization and algorithmic thinking applicable to business analytics.

// Sample C++ Code: Data Matching & Analysis for Ride-Sharing
#include <vector>
#include <algorithm>

// Data structure for driver information
struct Driver {
    int id;
    int capacity;
    bool accessible;
    double rating;
};

// Data structure for passenger information  
struct Passenger {
    int id;
    int needs;  // Number of passengers
    bool requiresAccessible;
};

// Matching algorithm based on data analysis
vector<int> matchDriversToPassengers(vector<Driver>& drivers, vector<Passenger>& passengers) {
    vector<int> matches;
    
    // Sort by capacity for optimal matching (data optimization)
    sort(drivers.begin(), drivers.end(), 
         [](Driver a, Driver b) { return a.capacity > b.capacity; });
    
    // Match passengers with appropriate drivers
    for (auto& passenger : passengers) {
        for (auto& driver : drivers) {
            if (driver.capacity >= passenger.needs && 
                (!passenger.requiresAccessible || driver.accessible)) {
                matches.push_back(driver.id);
                driver.capacity -= passenger.needs;  // Update available capacity
                break;
            }
        }
    }
    
    // Calculate and report matching success rate (analytics)
    double successRate = (matches.size() * 100.0) / passengers.size();
    cout << "Matching success rate: " << successRate << "%" << endl;
    
    return matches;
}
